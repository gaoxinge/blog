def __getattribute__(self, item):

    def class_lookup(cls, item):
        v = cls.__dict__.get(item)
        if v is not None:
            return v, cls
        for x in cls.__bases__:
            v, c = class_lookup(x, item)
            if v is not None:
                return v, c
        return None, None

    v, cls = class_lookup(type(self), item)
    if (v is not None) and hasattr(v, '__get__') and hasattr(v, '__set__'):
        return v.__get__(self, cls)
    w = self.__dict__.get(item)
    if w is not None:
        return w
    if v is not None:
        if hasattr(v, '__get__'):
            return v.__get__(self, cls)
        else:
            return v
    raise AttributeError(type(self), item)

特殊的描述符

import re

class UnboundMethod(object):

    def __init__(self, f):
        self.f = f

    def __call__(self, *args, **kwargs):
        return self.f(*args, **kwargs)

    def __str__(self):
        return '<unbound method %s>' % self.f.__name__

    __repr__ = __str__

class BoundMethod(object):

    def __init__(self, f, obj):
        self.f = f
        self.obj = obj

    def __call__(self, *args, **kwargs):
        return self.f(self.obj, *args, **kwargs)

    def __str__(self):
        return '<bound method %s of %s>' % (self.f.__name__, self.obj)

    __repr__ = __str__

class InstanceMethod(object):

    def __init__(self, f):
        self.f = f

    def __get__(self, obj, typ=None):
        if obj is None:
            return UnboundMethod(self.f)
        else:
            return BoundMethod(self.f, obj)

class ClassMethod(object):

    def __init__(self, f):
        self.f = f

    def __get__(self, obj, typ=None):
        return BoundMethod(self.f, typ)

class StaticMethod(object):

    def __init__(self, f):
        self.f = f

    def __get__(self, obj, typ=None):
        return self.f

class Property(object):

    def __init__(self, fget=None, fset=None, fdel=None, doc=None):
        self.fget = fget
        self.fset = fset
        self.fdel = fdel

        if not doc and not fget:
            doc = fget.__doc__

        self.__doc__ = doc

    def __get__(self, obj, typ=None):
        if not obj:
            return self
        if not self.fget:
            raise AttributeError('unreadable attribute')
        return self.fget(obj)

    def __set__(self, obj, value):
        if not self.fset:
            raise AttributeError('can\'t set attribute')
        self.fset(obj, value)

    def __delete__(self, obj):
        if not self.fdel:
            raise AttributeError('can\'t delete attribute')
        self.fdel(obj)

    def getter(self, fget):
        return self.__class__(fget, self.fset, self.fdel, self.__doc__)

    def setter(self, fset):
        return self.__class__(self.fget, fset, self.fdel, self.__doc__)

    def deleter(self, fdel):
        return self.__class__(self.fget, self.fset, fdel, self.__doc__)

class Animal(object):

    def __init__(self, name):
        self.name = name

    @InstanceMethod
    def getName(self):
        return self.name

class Person(object):

    kind = 'person'

    @ClassMethod
    def getKind(cls):
        return cls.kind

class Child(object):

    @StaticMethod
    def hello():
        print 'hello'

class Student(object):

    def getEmail(self):
        if '_email' not in self.__dict__:
            raise Exception('no email')
        return self._email

    def setEmail(self, value):
        result = re.match('\w+@\w+\.\w+', value)
        if not result:
            raise Exception('not valid')
        self._email = value

    def delEmail(self):
        del self._email

    email = property(getEmail, setEmail, delEmail, 'email property')
